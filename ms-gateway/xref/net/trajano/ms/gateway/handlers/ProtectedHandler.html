44  @Component
45  @Order(SelfRegisteringRoutingContextHandler.CORE_PATHS + 5)
46  public class ProtectedHandler extends SelfRegisteringRoutingContextHandler {
47  
48      private static final String BEARER_TOKEN_PATTERN = "^Bearer ([A-Za-z0-9]{64})$";
49  
50      private static final String INTERNAL_SERVER_ERROR = "Internal Server Error";
51  
52      private static final Logger LOG = LoggerFactory.getLogger(ProtectedHandler.class);
53  
54      private static final String X_JWKS_URI = "X-JWKS-URI";
55  
56      private static final String X_JWT_ASSERTION = "X-JWT-Assertion";
57  
58      private static final String X_JWT_AUDIENCE = "X-JWT-Audience";
59  
60      @Value("${authorization.endpoint}")
61      private URI authorizationEndpoint;
62  
63      @Autowired
64      private GatewayClientAuthorization gatewayClientAuthorization;
65  
66      @Autowired
67      private HttpClient httpClient;
68  
69      @Value("${jwks.path}")
70      private String jwksPath;
71  
72      @Value("${jwks.uri}")
73      private URI jwksUri;
74  
75      /**
76       * Obtains the access token from the request. It is expected to be the
77       * Authorization with a bearer tag. The authentication code is expected to be a
78       * given pattern.
79       *
80       * @param contextRequest
81       *            request
82       * @return access token
83       */
84      private String getAccessToken(final HttpServerRequest contextRequest) {
85  
86          final String authorizationHeader = contextRequest.getHeader(AUTHORIZATION);
87          if (authorizationHeader == null) {
88              return null;
89          }
90          final Matcher m = Pattern.compile(BEARER_TOKEN_PATTERN).matcher(authorizationHeader);
91          if (!m.matches()) {
92              return null;
93          } else {
94              return m.group(1);
95          }
96      }
97  
98      /**
99       * This handler goes through the authorization to prepopulate the
100      * X-JWT-Assertion header. There is no need to check if client is authorized
101      * because it was already done by ClientOriginHandler.
102      *
103      * @return handler
104      */
105     @Override
106     @SuppressWarnings("unchecked")
107     public void handle(final RoutingContext context) {
108 
109         LOG.debug("Handling {} ended={}", context, context.request().isEnded());
110         // if null or it is marked as protected then go on.
111         final PathContext pathContext = PathContext.get(context);
112         if (pathContext == null || !pathContext.isProtected()) {
113             context.next();
114             return;
115         }
116         LOG.debug("Handling {} with from={} to={} protected={}", context, pathContext.getFrom(), pathContext.getTo(), pathContext.isProtected());
117 
118         final String baseUri = pathContext.getFrom();
119 
120         final HttpServerRequest contextRequest = context.request();
121         final HttpServerResponse contextResponse = context.response();
122 
123         if (!contextRequest.uri().startsWith(baseUri)) {
124             throw new IllegalStateException(contextRequest.uri() + " did not start with" + baseUri);
125         }
126 
127         final String accessToken = getAccessToken(contextRequest);
128 
129         final String requestID = context.get(RequestIDProvider.REQUEST_ID);
130         final String now = RFC_1123_DATE_TIME.format(now(UTC));
131 
132         if (accessToken == null) {
133             LOG.debug("missing or invalid access token");
134             contextResponse
135                 .setStatusCode(401)
136                 .setStatusMessage("Unauthorized")
137                 .putHeader("WWW-Authenticate", "Bearer")
138                 .putHeader(HttpHeaders.CONTENT_TYPE, APPLICATION_JSON)
139                 .end(Errors.unauthorizedClient("missing or invalid access token").toBuffer());
140             return;
141         }
142 
143         final HttpClientRequest authorizationRequest = httpClient
144             .post(Conversions.toRequestOptions(authorizationEndpoint.resolve("/token")))
145             .handler(authorizationResponse -> {
146                 LOG.debug("ended={}", contextRequest.isEnded());
147                 authorizationResponse.bodyHandler(buffer -> {
148                     LOG.debug("isended={}", contextRequest.isEnded());
149                     if (authorizationResponse.statusCode() != 200) {
150                         StreamSupport.stream(authorizationResponse.headers().spliterator(), false)
151                             .filter(Predicates.STRIP_CONTENT_LENGTH_AND_TRANSFER_ENCODING)
152                             .forEach(h -> contextResponse.putHeader(h.getKey(), h.getValue()));
153                         contextResponse.setStatusCode(authorizationResponse.statusCode())
154                             .setStatusMessage(authorizationResponse.statusMessage())
155                             .setChunked(false)
156                             .end(buffer);
157                     } else {
158                         final JsonObject response = new JsonObject(buffer);
159                         final String idToken = response.getString("id_token");
160 
161                         final List<String> audience = response.getJsonArray("aud").getList();
162                         if (idToken == null) {
163                             LOG.error("Unable to get the ID Token from {} given access_token={}", authorizationEndpoint, accessToken);
164                             context.response().setStatusCode(500)
165                                 .setStatusMessage(INTERNAL_SERVER_ERROR)
166                                 .putHeader(CONTENT_TYPE, APPLICATION_JSON)
167                                 .end(Errors.serverError("Unable to get assertion from authorization endpoint").toBuffer());
168                             return;
169                         }
170 
171                         final Map<String, String> securityMap = new HashMap<>();
172                         securityMap.put(X_JWT_ASSERTION, idToken);
173                         securityMap.put(X_JWKS_URI, jwksUri.toASCIIString());
174                         securityMap.put(X_JWT_AUDIENCE, audience.stream().collect(Collectors.joining(", ")));
175                         context.put("additional_headers", securityMap);
176                         contextRequest.resume();
177                         context.next();
178                     }
179                 });
180 
181             })
182             .exceptionHandler(context::fail);
183 
184         gatewayClientAuthorization.gatewayRequest(authorizationRequest)
185             .putHeader(HttpHeaders.CONTENT_TYPE, APPLICATION_FORM_URLENCODED)
186             .putHeader(HttpHeaders.ACCEPT, APPLICATION_JSON)
187             .putHeader(REQUEST_ID, requestID)
188             .putHeader(DATE, now)
189             .end("grant_type=authorization_code&code=" + accessToken);
190         contextRequest.pause();
191         LOG.debug("access_token={} ended={}", accessToken, contextRequest.isEnded());
192     }
193 
194     @Override
195     public void register(final Router router) {
196 
197         router.get().handler(this);
198         router.post().handler(this);
199         router.put().handler(this);
200         router.delete().handler(this);
201     }
