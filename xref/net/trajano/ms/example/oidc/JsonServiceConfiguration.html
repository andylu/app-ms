3   import java.io.File;
4   import java.net.URI;
5   import java.util.Map;
6   import java.util.function.Function;
7   import java.util.stream.Collectors;
8   
9   import javax.annotation.PostConstruct;
10  import javax.inject.Inject;
11  import javax.ws.rs.client.Client;
12  import javax.ws.rs.client.ClientBuilder;
13  import javax.ws.rs.core.MediaType;
14  import javax.ws.rs.core.UriBuilder;
15  
16  import org.slf4j.Logger;
17  import org.slf4j.LoggerFactory;
18  import org.springframework.beans.factory.annotation.Value;
19  import org.springframework.context.annotation.Configuration;
20  
21  import net.trajano.ms.core.JsonOps;
22  import net.trajano.ms.oidc.OpenIdConfiguration;
23  import net.trajano.ms.oidc.spi.IssuerConfig;
24  import net.trajano.ms.oidc.spi.IssuersConfig;
25  import net.trajano.ms.oidc.spi.ServiceConfiguration;
26  
27  @Configuration
28  public class JsonServiceConfiguration implements
29      ServiceConfiguration {
30  
31      private static final Logger LOG = LoggerFactory.getLogger(JsonServiceConfiguration.class);
32  
33      private Map<String, IssuerConfig> issuers;
34  
35      @Inject
36      private JsonOps jsonOps;
37  
38      @Value("${oidc.config_file:openidconnect-config.json}")
39      private File oidcFile;
40  
41      @Value("${oidc.redirect_uri}")
42      private URI redirectUri;
43  
44      @Value("${token_endpoint:}")
45      private URI tokenEndpoint;
46  
47      @Override
48      public IssuerConfig getIssuerConfig(final String issuerId) {
49  
50          return issuers.get(issuerId);
51      }
52  
53      @Override
54      public URI getRedirectUri() {
55  
56          return redirectUri;
57      }
58  
59      @PostConstruct
60      public void init() {
61  
62          final Client client = ClientBuilder.newClient();
63  
64          final IssuersConfig issuersConfig = jsonOps.fromJson(oidcFile, IssuersConfig.class);
65  
66          issuersConfig.getIssuers().forEach(issuer -> {
67              issuer.setOpenIdConfiguration(client.target(UriBuilder.fromUri(issuer.getUri()).path("/.well-known/openid-configuration")).request(MediaType.APPLICATION_JSON).get(OpenIdConfiguration.class));
68              LOG.info("Registered {} to {}", issuer.getId(), issuer.getUri());
69          });
70          issuers = issuersConfig.getIssuers().stream()
71              .collect(Collectors.toMap(IssuerConfig::getId, Function.identity()));
72  
73      }
74  
75  }
