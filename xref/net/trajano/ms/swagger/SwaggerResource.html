3   import javax.annotation.security.PermitAll;
4   import javax.ws.rs.GET;
5   import javax.ws.rs.NotFoundException;
6   import javax.ws.rs.Path;
7   import javax.ws.rs.PathParam;
8   import javax.ws.rs.Produces;
9   import javax.ws.rs.core.Context;
10  import javax.ws.rs.core.MediaType;
11  import javax.ws.rs.core.UriInfo;
12  
13  import org.springframework.beans.factory.annotation.Autowired;
14  
15  import io.swagger.annotations.Api;
16  import io.swagger.models.Swagger;
17  import net.trajano.ms.swagger.internal.SwaggerCollator;
18  
19  @Api
20  @Path("/")
21  @PermitAll
22  public class SwaggerResource {
23  
24      @Autowired
25      private SwaggerCollator collator;
26  
27      @GET
28      @Produces(MediaType.APPLICATION_JSON)
29      @Path("/{version}")
30      public Swagger swagger(@PathParam("version") final String version,
31          @Context final UriInfo uriInfo) {
32  
33          if (!collator.isPathExists(version)) {
34              throw new NotFoundException();
35          }
36          return collator.getSwagger(version, uriInfo);
37      }
38  }
