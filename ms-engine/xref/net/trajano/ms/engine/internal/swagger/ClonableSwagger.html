1   package net.trajano.ms.engine.internal.swagger;
2   
3   import java.util.Collections;
4   
5   import com.fasterxml.jackson.annotation.JsonInclude;
6   import com.fasterxml.jackson.annotation.JsonInclude.Include;
7   
8   import io.swagger.models.Scheme;
9   import io.swagger.models.Swagger;
10  import io.vertx.ext.web.RoutingContext;
11  
12  @JsonInclude(Include.NON_NULL)
13  public class ClonableSwagger extends Swagger {
14  
15      /**
16       * Creates a shallow copy of the swagger but set the schemes, host and base path
17       * based on {@link RoutingContext}.
18       *
19       * @param context
20       *            routing context
21       * @return shallow copy
22       */
23      public ClonableSwagger withRoutingContext(final RoutingContext context) {
24  
25          final ClonableSwagger copy = new ClonableSwagger();
26          copy.swagger = swagger;
27          copy.info = info;
28          copy.host = context.request().host();
29          copy.basePath = context.currentRoute().getPath();
30          copy.tags = tags;
31  
32          copy.schemes = Collections.singletonList(Scheme.forValue(context.request().scheme()));
33  
34          copy.consumes = consumes;
35          copy.produces = produces;
36          copy.security = security;
37          copy.paths = paths;
38          copy.securityDefinitions = securityDefinitions;
39          copy.definitions = definitions;
40          copy.parameters = parameters;
41          copy.responses = responses;
42          copy.externalDocs = externalDocs;
43          copy.vendorExtensions = vendorExtensions;
44          return copy;
45      }
46  
47  }
