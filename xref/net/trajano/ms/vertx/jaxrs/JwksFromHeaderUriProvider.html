1   package net.trajano.ms.vertx.jaxrs;
2   
3   import java.net.URI;
4   
5   import javax.ws.rs.container.ContainerRequestContext;
6   
7   import org.slf4j.Logger;
8   import org.slf4j.LoggerFactory;
9   import org.springframework.beans.factory.annotation.Autowired;
10  import org.springframework.beans.factory.annotation.Qualifier;
11  import org.springframework.context.annotation.Configuration;
12  import org.springframework.stereotype.Component;
13  
14  @Component
15  @Configuration
16  public class JwksFromHeaderUriProvider implements
17      JwksUriProvider {
18  
19      private static final Logger LOG = LoggerFactory.getLogger(JwksFromHeaderUriProvider.class);
20  
21      @Autowired(required = false)
22      @Qualifier("authz.signature.jwks.uri")
23      private URI signatureJwksUri;
24  
25      @Override
26      public URI getUri(final ContainerRequestContext requestContext) {
27  
28          if (signatureJwksUri != null) {
29              return signatureJwksUri;
30          }
31          final String jwksUriFromHeader = requestContext.getHeaderString("X-JWKS-URI");
32          if (jwksUriFromHeader != null) {
33              return URI.create(jwksUriFromHeader);
34          } else {
35              LOG.warn("Neither `authz.signature.jwks.uri` was specified nor `X-JWKS-URI` was found in the header, signature verification will not be performed");
36              return null;
37          }
38      }
39  }
