37  /**
38   * This works like the FORM based login of Java EE. It allows any user name as
39   * long as the password is "password"
40   */
41  @SwaggerDefinition(
42      info = @Info(
43          title = "Sample Authn Microservice",
44          version = "1.0"))
45  @Api
46  @Component
47  @Path("/authn")
48  @PermitAll
49  public class AuthnResource {
50  
51      @Value("${authorizationEndpoint}")
52      private URI authorizationEndpoint;
53  
54      @Context
55      private Client client;
56  
57      @Autowired
58      private CryptoOps cryptoOps;
59  
60      @POST
61      @Consumes(MediaType.APPLICATION_JSON)
62      @Produces(MediaType.APPLICATION_JSON)
63      @ApiResponses(@ApiResponse(code = 401,
64          message = "Unauthorized Response",
65          response = ErrorResponse.class))
66      public Response json(final UsernamePassword usernamePassword,
67          @HeaderParam(HttpHeaders.AUTHORIZATION) final String authorization) {
68  
69          if (!"password".equals(usernamePassword.getPassword())) {
70              throw ErrorResponses.unauthorized(ErrorCodes.UNAUTHORIZED_CLIENT, "invalid username/password combination", "FORM");
71          }
72  
73          final JwtClaims claims = new JwtClaims();
74          claims.setSubject(usernamePassword.getUsername());
75          claims.setAudience(HttpAuthorizationHeaders.parseBasicAuthorization(authorization)[0]);
76  
77          final Form form = new Form();
78          form.param("grant_type", GrantTypes.JWT_ASSERTION);
79          form.param("assertion", cryptoOps.sign(claims));
80  
81          return Response.ok(client.target(authorizationEndpoint).request(MediaType.APPLICATION_JSON_TYPE)
82              .header(HttpHeaders.AUTHORIZATION, authorization)
83              .post(Entity.form(form), OAuthTokenResponse.class))
84              .build();
85  
86      }
87  
88  }
