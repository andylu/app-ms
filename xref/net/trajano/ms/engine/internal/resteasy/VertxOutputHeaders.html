3   import java.util.AbstractMap;
4   import java.util.HashMap;
5   import java.util.LinkedList;
6   import java.util.List;

8   import java.util.Set;
9   import java.util.stream.Collectors;
10  
11  import javax.ws.rs.core.AbstractMultivaluedMap;
12  
13  import io.vertx.core.http.HttpServerResponse;
14  
15  public class VertxOutputHeaders extends AbstractMultivaluedMap<String, Object> {
16  
17      private static class VertxOutputHeadersMap extends AbstractMap<String, List<Object>> {
18  
19          private final HttpServerResponse vertxResponse;
20  
21          public VertxOutputHeadersMap(final HttpServerResponse vertxResponse) {
22  
23              this.vertxResponse = vertxResponse;
24  
25          }
26  
27          @Override
28          public Set<Entry<String, List<Object>>> entrySet() {
29  
30              final Map<String, List<Object>> ret = new HashMap<>();
31              vertxResponse.headers().forEach(entry -> {
32                  final List<Object> list = ret.getOrDefault(entry.getKey(), new LinkedList<>());
33                  ret.put(entry.getKey(), list);
34              });
35              return ret.entrySet();
36          }
37  
38          @Override
39          public boolean equals(final Object o) {
40  
41              return super.equals(o);
42          }
43  
44          @Override
45          public int hashCode() {
46  
47              return super.hashCode();
48          }
49  
50          @Override
51          public List<Object> put(final String key,
52              final List<Object> values) {
53  
54              final List<Object> prev = vertxResponse.headers().getAll(key).stream().map(v -> v).collect(Collectors.toList());
55              final List<String> collect = values.stream().map(String::valueOf).collect(Collectors.toList());
56              vertxResponse.putHeader(key, collect);
57              return prev;
58          }
59      }
60  
61      public VertxOutputHeaders(final HttpServerResponse vertxResponse) {
62  
63          super(new VertxOutputHeadersMap(vertxResponse));
64      }
65  }
