54      private static final Logger LOG = LoggerFactory.getLogger(TokenResource.class);
55

83      /**
84       * Performs client credential validation then dispatches to the appropriate
85       * handler for a given grant type.
86       */
87      @POST
88      @Consumes(MediaType.APPLICATION_FORM_URLENCODED)
89      @Produces(MediaType.APPLICATION_JSON)
90      public OAuthTokenResponse dispatch(
91          @ApiParam(allowableValues = "refresh_token, authorization_code") @FormParam("grant_type") final String grantType,
92          @FormParam("code") final String code,
93          @FormParam("assertion") final String assertion,
94          @FormParam("aud") final String audience,
95          @FormParam("refresh_token") final String refreshToken,
96          @FormParam("jwks_uri") final URI jwksUri,
97          @HeaderParam(HttpHeaders.AUTHORIZATION) final String authorization) {
98  
99          final String[] clientCredentials = HttpAuthorizationHeaders.parseBasicAuthorization(authorization);
100 
101         final String clientId = clientCredentials[0];
102         if (!clientValidator.isValid(grantType, clientId, clientCredentials[1])) {
103             throw ErrorResponses.unauthorized(ErrorCodes.UNAUTHORIZED_CLIENT, "Unauthorized client", String.format("Basic realm=\"%s\", encoding=\"UTF-8\"", realmName));
104         }
105 
106         if (GrantTypes.REFRESH_TOKEN.equals(grantType)) {
107             return handleRefreshGrant(refreshToken, clientId);
108         } else if (GrantTypes.AUTHORIZATION_CODE.equals(grantType)) {
109             return handleAuthorizationCodeGrant(code);
110         } else if (GrantTypes.JWT_ASSERTION.equals(grantType)) {
111             return handleJwtAssertionGrant(assertion, clientId, audience);
112 
113         } else {
114             throw ErrorResponses.badRequest(ErrorCodes.UNSUPPORT_GRANT_TYPE, "Invalid grant type");
115         }
116 
117     }
118 
119     private IdTokenResponse handleAuthorizationCodeGrant(final String accessToken) {
120 
121         if (accessToken == null) {
122             throw ErrorResponses.invalidRequest("Missing access token");
123         }
124         final IdTokenResponse idTokenResponse = tokenCache.get(accessToken);
125         if (idTokenResponse == null) {
126             throw ErrorResponses.unauthorized(ErrorCodes.UNAUTHORIZED_CLIENT, "Access token was not valid", "Bearer");
127         }
128         return idTokenResponse;
129 
130     }
131 
132     /**
133      * Takes an assertion and converts it using an {@link InternalClaimsBuilder} to
134      * a JWT used internally
135      *
136      * @param assertion
137      *            an external JWT assertion
138      * @param clientId
139      *            client ID
140      * @return OAuth response
141      */
142     private OAuthTokenResponse handleJwtAssertionGrant(final String assertion,
143         final String clientId,
144         final String audience) {
145 
146         if (assertion == null) {
147             throw ErrorResponses.badRequest(ErrorCodes.INVALID_REQUEST, "Missing assertion");
148         }
149         if (clientId == null) {
150             throw ErrorResponses.badRequest(ErrorCodes.INVALID_REQUEST, "Missing client_id");
151         }
152 
153         try {
154             final URI jwksUri = clientValidator.getJwksUri(clientId);
155             LOG.debug("jwksUri={}", jwksUri);
156             HttpsJwks httpsJwks = null;
157             if (jwksUri != null) {
158                 httpsJwks = jwksMap.computeIfAbsent(jwksUri, uri -> new HttpsJwks(uri.toASCIIString()));
159             }
160 
161             final JwtConsumerBuilder builder = new JwtConsumerBuilder();
162 
163             if (httpsJwks == null) {
164                 builder.setDisableRequireSignature()
165                     .setSkipSignatureVerification();
166             } else {
167                 builder.setVerificationKeyResolver(new HttpsJwksVerificationKeyResolver(httpsJwks));
168             }
169             if (audience == null) {
170                 builder.setExpectedAudience(clientId);
171             } else {
172                 builder.setExpectedAudience(clientId, audience);
173             }
174             final JwtConsumer jwtConsumer = builder
175                 .build();
176 
177             final JwtClaims internalClaims = internalClaimsBuilder.buildInternalJWTClaimsSet(jwtConsumer.processToClaims(assertion));
178 
179             if (internalClaims.getSubject() == null) {
180                 LOG.error("Subject is missing from {}", internalClaims);
181                 throw ErrorResponses.internalServerError("Subject is missing from the resulting claims set.");
182             }
183 
184             internalClaims.setGeneratedJwtId();
185             internalClaims.setIssuer(issuer.toASCIIString());
186             if (audience == null) {
187                 internalClaims.setAudience(clientId);
188             } else {
189                 internalClaims.setAudience(clientId, audience);
190             }
191             internalClaims.setIssuedAtToNow();
192 
193             final Instant expirationTime = Instant.now().plus(jwtMaximumLifetimeInSeconds, ChronoUnit.SECONDS);
194             internalClaims.setExpirationTime(NumericDate.fromMilliseconds(expirationTime.toEpochMilli()));
195 
196             return tokenCache.store(cryptoOps.sign(internalClaims), internalClaims.getAudience(), expirationTime);
197 
198         } catch (final MalformedClaimException
199             | InvalidJwtException e) {
200             LOG.error("Unable to parse assertion", e);
201             throw ErrorResponses.badRequest(ErrorCodes.INVALID_REQUEST, "Unable to parse assertion");
202         }
203     }
204 
205     private OAuthTokenResponse handleRefreshGrant(final String refreshToken,
206         final String clientId) {
207 
208         if (refreshToken == null) {
209             throw ErrorResponses.badRequest(ErrorCodes.INVALID_REQUEST, "Missing refresh token");
210         }
211         return tokenCache.refresh(refreshToken, clientId);
212 
213     }
