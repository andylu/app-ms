1   package net.trajano.ms.vertx.beans;
2   
3   import java.lang.reflect.Method;
4   
5   import javax.annotation.security.PermitAll;
6   import javax.annotation.security.RolesAllowed;
7   import javax.ws.rs.container.ResourceInfo;
8   
9   import org.slf4j.Logger;
10  import org.slf4j.LoggerFactory;
11  
12  /**
13   * Checks if the URI has any JSR-250 annotations that would indicate a secured
14   * resource. On a method level if PermitAll is present then it will allow access
15   * without any assertion even if RolesAllowed is present on the class level.
16   * <p>
17   * However, the default is to require the assertion if nothing has been
18   * specified.
19   * <p>
20   * This will throw an IllegalArgumentException if both {@link RolesAllowed} and
21   * {@link PermitAll} are present in a class or method.
22   *
23   * @author Archimedes Trajano
24   */
25  public class DefaultAssertionRequiredPredicate implements
26      JwtAssertionRequiredPredicate {
27  
28      private static final Logger LOG = LoggerFactory.getLogger(DefaultAssertionRequiredPredicate.class);
29  
30      /**
31       * <p>
32       * The key logic for the test is as follows:
33       * </p>
34       * <p>
35       * Let:
36       * </p>
37       * <table>
38       * <tr>
39       * <td>a =</td>
40       * <td>resourceMethodHasRolesAllowed</td>
41       * </tr>
42       * <tr>
43       * <td>b =</td>
44       * <td>resourceClassHasRolesAllowed</td>
45       * </tr>
46       * <tr>
47       * <td>c =</td>
48       * <td>resourceMethodHasPermitAll</td>
49       * </tr>
50       * <tr>
51       * <td>d =</td>
52       * <td>resourceClassHasPermitAll</td>
53       * </tr>
54       * </table>
55       * <p>
56       * The rules that need to be applied translate to:
57       * </p>
58       *
59       * <pre>
60       * = a || ( b && !c ) || ( !a && !b && !c && !d )
61       * = (a || b || !d) && ( a || !c )
62       * </pre>
63       *
64       * @param resourceInfo
65       *            resource info
66       * @return <code>true</code> if the resource is protected
67       */
68      @Override
69      public boolean test(final ResourceInfo resourceInfo) {
70  
71          final Method resourceMethod = resourceInfo.getResourceMethod();
72          final Class<?> resourceClass = resourceInfo.getResourceClass();
73  
74          final boolean resourceMethodHasRolesAllowed = resourceMethod.getAnnotation(RolesAllowed.class) != null;
75          final boolean resourceClassHasRolesAllowed = resourceClass.getAnnotation(RolesAllowed.class) != null;
76  
77          final boolean resourceMethodHasPermitAll = resourceMethod.getAnnotation(PermitAll.class) != null;
78          final boolean resourceClassHasPermitAll = resourceClass.getAnnotation(PermitAll.class) != null;
79  
80          if (LOG.isDebugEnabled()) {
81              LOG.debug("resourceMethod={} PermitAll={} RolesAllowed={}", resourceMethod, resourceMethodHasPermitAll, resourceMethodHasRolesAllowed);
82              LOG.debug("resourceClass={} PermitAll={} RolesAllowed={}", resourceClass, resourceClassHasPermitAll, resourceClassHasRolesAllowed);
83          }
84  
85          if (resourceMethodHasRolesAllowed && resourceMethodHasPermitAll) {
86              throw new IllegalArgumentException("The resource method " + resourceMethod + " may not have both @RolesAllowed and @PermitAll annotations.");
87          } else if (resourceClassHasRolesAllowed && resourceClassHasPermitAll) {
88              throw new IllegalArgumentException("The resource class " + resourceClass + " may not have both @RolesAllowed and @PermitAll annotations.");
89          } else {
90              return (resourceMethodHasRolesAllowed || resourceClassHasRolesAllowed || !resourceClassHasPermitAll) && (resourceMethodHasRolesAllowed || !resourceMethodHasPermitAll);
91          }
92  
93      }
94  
95  }
