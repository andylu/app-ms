9       /**
10       * Host name and port.

14      /**
15       * JWT ID. Represents a client session;
16       */
17      public static final String JWT_ID = "X-JWT-ID";
18  
19      /**
20       * Request ID. Represents a single request coming from the client.
21       */
22      public static final String REQUEST_ID = "X-Request-ID";
23  
24      /**
25       * Request method. Represents the method URI being requested upon.
26       */
27      public static final String REQUEST_METHOD = "X-Request-Method";

29      /**
30       * Request URI. Represents the URI being requested.
31       */
32      public static final String REQUEST_URI = "X-Request-URI";
