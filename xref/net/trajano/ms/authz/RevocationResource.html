29  /**
30   * Revocation endpoint resource.
31   *
32   * @author Archimedes Trajano
33   */
34  @Api
35  @Configuration
36  @Component
37  @Path("/revoke")
38  @PermitAll
39  public class RevocationResource {
40  
41      private static final Logger LOG = LoggerFactory.getLogger(RevocationResource.class);
42  
43      @Autowired
44      private ClientValidator clientValidator;
45  
46      /**
47       * Maximum life of a JWT token. Past that period, it is expected to no longer be
48       * used.
49       */
50      @Value("${token.jwtMaximumLifetime:86400}")
51      private int jwtMaximumLifetimeInSeconds;
52  
53      @Value("${realmName:client_credentials}")
54      private String realmName;
55  
56      @Autowired
57      private TokenCache tokenCache;
58  
59      /**
60       * Performs client credential validation then invokes the revocation process
61       * from the token cache.
62       */
63      @POST
64      @Consumes(MediaType.APPLICATION_FORM_URLENCODED)
65      @Produces(MediaType.APPLICATION_JSON)
66      public JsonObject revoke(
67          @FormParam("token") final String token,
68          @FormParam("token_type_hint") final String tokenTypeHint,
69          @HeaderParam(HttpHeaders.AUTHORIZATION) final String authorization) {
70  
71          final String[] clientCredentials = HttpAuthorizationHeaders.parseBasicAuthorization(authorization);
72  
73          final String clientId = clientCredentials[0];
74          if (!clientValidator.isValid("revocation", clientId, clientCredentials[1])) {
75              throw ErrorResponses.unauthorized(ErrorCodes.UNAUTHORIZED_CLIENT, "Unauthorized client", String.format("Basic realm=\"%s\", encoding=\"UTF-8\"", realmName));
76          }
77  
78          if (token == null) {
79              throw ErrorResponses.invalidRequest("Missing token");
80          }
81          if (tokenTypeHint != null && !"refresh_token".equals(tokenTypeHint)) {
82              throw ErrorResponses.badRequest(ErrorCodes.UNSUPPORTED_TOKEN_TYPE, "Token type is not supported");
83          }
84  
85          tokenCache.revokeRefreshToken(token, clientId);
86          LOG.debug("revoked token={}", token);
87  
88          final JsonObject okReturn = new JsonObject();
89          okReturn.addProperty("ok", 1);
90          return okReturn;
91  
92      }
93  
94  }
