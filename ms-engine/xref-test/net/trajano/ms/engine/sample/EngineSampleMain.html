19  @Configuration
20  public class EngineSampleMain extends AbstractVerticle {
21  
22      public static void main(final String[] args) throws Exception {
23  
24          final VertxOptions vertOptions = new VertxOptions();
25          vertOptions.setWarningExceptionTime(1);
26          vertOptions.setWorkerPoolSize(50);
27  
28          final Vertx vertx = Vertx.vertx(vertOptions);
29          final DeploymentOptions options = new DeploymentOptions();
30          vertx.deployVerticle(new EngineSampleMain(), options);
31      }
32  
33      private SpringJaxRsHandler requestHandler;
34  
35      @Bean
36      public HttpClientOptions httpClientOptions() {
37  
38          return new HttpClientOptions();
39      }
40  
41      @Override
42      public void start() throws Exception {
43  
44          final Router router = Router.router(vertx);
45  
46          final HttpServerOptions httpServerOptions = new HttpServerOptions();
47          httpServerOptions.setPort(8900);
48  
49          final HttpServer http = vertx.createHttpServer(httpServerOptions);
50  
51          SwaggerHandler.registerToRouter(router, MyApp.class);
52          final JaxRsRouter jaxRsRouter = new JaxRsRouter();
53          final SpringJaxRsHandler handler = new SpringJaxRsHandler(MyApp.class);
54          jaxRsRouter.register(MyApp.class, router, handler, handler);
55          ManifestHandler.registerToRouter(router);
56  
57          http.requestHandler(req -> router.accept(req)).listen(res -> {
58              if (res.failed()) {
59                  res.cause().printStackTrace();
60                  vertx.close();
61              }
62          });
63      }
64  
65      @Override
66      public void stop() throws Exception {
67  
68          requestHandler.close();
69      }
70  
71  }
