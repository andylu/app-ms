28  /**
29   * This resource uses CDI rather than Spring annotations.
30   *
31   * @author Archimedes Trajano
32   */
33  @Api(tags = {
34      "infernal",
35      "doom"
36  },
37      authorizations = @Authorization("Bearer"))
38  @ApplicationScoped
39  @Path("/s")
40  public class SecureHelloResource {
41  
42      @Inject
43      private Counter counter;
44  
45      @Inject
46      private JsonOps jsonOps;
47  
48      @Inject
49      private UselessCounter uselessCounter;
50  
51      @ApiOperation(value = "Counts")
52      @GET
53      @Produces(MediaType.APPLICATION_JSON)
54      @Path("/count")
55      public JsonObject count() {
56  
57          final JsonObject json = new JsonObject();
58          json.add("counter", new JsonPrimitive(counter.count()));
59          json.add("useless_counter", new JsonPrimitive(uselessCounter.count()));
60          return json;
61      }
62  
63      @ApiOperation(value = "Secure Hello")
64      @GET
65      @Produces(MediaType.APPLICATION_JSON)
66      public JsonObject hello(@QueryParam("name") final String name,
67          @Context final SecurityContext securityContext) {
68  
69          final JsonObject json = new JsonObject();
70          if (name != null) {
71              json.add("name", new JsonPrimitive(name));
72          }
73          final JwtClaimsSetPrincipal userPrincipal = (JwtClaimsSetPrincipal) securityContext.getUserPrincipal();
74          json.add("principal", new JsonPrimitive(userPrincipal.getName()));
75          json.add("claims", jsonOps.toJsonElement(userPrincipal.getClaimsSet().toJson()));
76          return json;
77      }
78  
79      @ApiOperation(value = "displays hello world after a given amount of seconds seconds")
80      @GET
81      @Path("/suspend/{seconds}")
82      @Produces(MediaType.TEXT_PLAIN)
83      public void suspend(@Suspended final AsyncResponse asyncResponse,
84          @PathParam("seconds") final int seconds) throws InterruptedException {
85  
86          Thread.sleep(seconds * 1000L);
87          asyncResponse.resume(Response.ok("hello").build());
88      }

90  }
