26  /**
27   * <p>
28   * Handles determining what Origin URIs are allowed based on the Origin header.
29   * This will make sure that the client is allowed based on the Origin to see if
30   * it is whitelisted.
31   * </p>
32   * This is done after ClientOriginHandler and checks if the client has already
33   * been authorized there using the context data.
34   *
35   * @author Archimedes Trajano
36   */
37  @Component
38  @Order(SelfRegisteringRoutingContextHandler.PROXIED)
39  public class UnauthenticatedClientValidator extends SelfRegisteringRoutingContextHandler {
40  
41      private static final Logger LOG = LoggerFactory.getLogger(UnauthenticatedClientValidator.class);
42  
43      @Value("${authorization.endpoint}")
44      private URI authorizationEndpoint;
45  
46      @Autowired
47      private GatewayClientAuthorization gatewayClientAuthorization;
48  
49      @Autowired
50      private HttpClient httpClient;
51  
52      @Override
53      public void handle(final RoutingContext context) {
54  
55          final PathContext pathContext = getPathContext(context);
56          if (pathContext == null || !pathContext.isProtected()) {
57              context.next();
58              return;
59          }
60  
61          // Check if the client is already authorized
62          final Boolean authorized = context.get("client_authorized");
63          LOG.debug("authorized={}", authorized);
64          if (Boolean.TRUE.equals(authorized)) {
65              context.next();
66              return;
67          }
68  
69          // The origin was already set by the previous step.
70          final URI originUri = context.get("origin");
71          LOG.debug("originUri={} clientCheckEndpoint={}", originUri, authorizationEndpoint);
72          final HttpClientRequest authorizationRequest = httpClient.post(Conversions.toRequestOptions(authorizationEndpoint.resolve("/check")), authorizationResponse -> {
73              LOG.debug("statusCode={}", authorizationResponse.statusCode());
74              if (authorizationResponse.statusCode() == 204) {
75                  context.next();
76              } else {
77                  authorizationResponse.headers().forEach(h -> stripTransferEncodingAndLength(context.response(), h));
78                  authorizationResponse.bodyHandler(body -> {
79                      context.response().setStatusCode(authorizationResponse.statusCode())
80                          .setStatusMessage(authorizationResponse.statusMessage())
81                          .setChunked(false)
82                          .end(body);
83                      LOG.debug("response={}", body);
84                  });
85              }
86          });
87          final JsonObject checkRequest = new JsonObject();
88          checkRequest.put("origin", originUri.toASCIIString());
89          LOG.debug("checkRequest={}", checkRequest);
90          gatewayClientAuthorization.gatewayRequest(authorizationRequest)
91              .putHeader(HttpHeaders.CONTENT_TYPE, APPLICATION_JSON)
92              .putHeader(HttpHeaders.ACCEPT, APPLICATION_JSON)
93              .end(checkRequest.toBuffer());
94          context.request().pause();
95  
96      }
97  
98      /**
99       * Register this only to the options() method handler. {@inheritDoc}
100      */
101     @Override
102     public void register(final Router router) {
103 
104         router.route().handler(this);
105 
106     }
107 
108     private void stripTransferEncodingAndLength(final HttpServerResponse contextResponse,
109         final Map.Entry<String, String> h) {
110 
111         if ("Content-Length".equalsIgnoreCase(h.getKey()) || "Transfer-Encoding".equalsIgnoreCase(h.getKey())) {
112             return;
113         }
114         contextResponse.putHeader(h.getKey(), h.getValue());
115     }
