1   package net.trajano.ms.core;
2   
3   import org.jose4j.jwk.HttpsJwks;
4   import org.jose4j.jwk.JsonWebKeySet;
5   import org.jose4j.jwt.JwtClaims;
6   
7   /**
8    * Common cryptography operations.
9    */
10  public interface CryptoOps {
11  
12      /**
13       * Generates a random string token suitable for non-durable cache lookups. Do
14       * not use these in place of UUIDs.
15       *
16       * @return string token
17       */
18      String newToken();
19  
20      /**
21       * Convert {@link JwtClaims} into a JWS.
22       *
23       * @param claims
24       *            claims
25       * @return JWS
26       */
27      String sign(JwtClaims claims);
28  
29      /**
30       * Get the claims set with a given JWT with no audience checking.
31       *
32       * @param jwt
33       *            JWT
34       * @param jwks
35       *            remote JWKS
36       * @return
37       */
38      default JwtClaims toClaimsSet(final String jwt,
39          final HttpsJwks jwks) {
40  
41          return toClaimsSet(jwt, null, jwks);
42      }
43  
44      /**
45       * @deprecated this method is not used.
46       * @return null
47       */
48      @Deprecated
49      default JwtClaims toClaimsSet(final String idToken,
50          final JsonWebKeySet jwks) {
51  
52          return null;
53      }
54  
55      /**
56       * Get the claims set with a given JWT.
57       *
58       * @param jwt
59       *            JWT
60       * @param audience
61       *            audience. If null specified then this will disable audience
62       *            checking.
63       * @param jwks
64       *            remote JWKS
65       * @return
66       */
67      JwtClaims toClaimsSet(String jwt,
68          String audience,
69          HttpsJwks jwks);
70  }
