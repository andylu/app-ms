89      public SpringJaxRsHandler(final ConfigurableApplicationContext baseApplicationContext,
90          final Class<? extends Application> applicationClass) {
91  
92          if (baseApplicationContext == null) {
93              LOG.debug("baseApplicationContext is null, a new application context will be created.");
94              applicationContext = new AnnotationConfigApplicationContext();
95          } else if (baseApplicationContext.isActive()) {
96              LOG.debug("baseApplicationContext={} is active, will use as parent.", baseApplicationContext);
97              applicationContext = new AnnotationConfigApplicationContext();
98              applicationContext.setParent(baseApplicationContext);
99          } else if (!(baseApplicationContext instanceof AnnotationConfigApplicationContext)) {
100             LOG.debug("baseApplicationContext={} is not active, but is not an instance of AnnotationConfigApplicationContext, will activate and use as parent.", baseApplicationContext);
101             baseApplicationContext.refresh();
102             applicationContext = new AnnotationConfigApplicationContext();
103             applicationContext.setParent(baseApplicationContext);
104         } else {
105             LOG.debug("baseApplicationContext={} is not active, will reuse.", baseApplicationContext);
106             applicationContext = (AnnotationConfigApplicationContext) baseApplicationContext;
107         }
108         applicationContext.setScopeMetadataResolver(new CdiScopeMetadataResolver());
109         applicationContext.register(SpringConfiguration.class, applicationClass, VertxRequestContextFilter.class, CommonObjectMapperProvider.class, WebApplicationExceptionMapper.class);
110 
111         final ApplicationPath annotation = applicationClass.getAnnotation(ApplicationPath.class);
112         if (annotation != null) {
113             baseUri = URI.create(annotation.value() + "").normalize();
114         } else {
115             baseUri = URI.create("");
116         }
117 
118         Application application;
119         try {
120             application = applicationClass.newInstance();
121         } catch (InstantiationException
122             | IllegalAccessException e) {
123             throw new ExceptionInInitializerError(e);
124         }
125 
126         pathAnnotatedClasses = new HashSet<>();
127         final Set<Class<?>> resourceClasses = application.getClasses();
128         if (resourceClasses.isEmpty()) {
129             final String packageName = applicationClass.getPackage().getName();
130             LOG.debug("getClasses() is empty, performing scanning of annotated JAX-RS classes from {}", packageName);
131             final Reflections reflections = new Reflections(packageName);
132             reflections.getTypesAnnotatedWith(Path.class).forEach(clazz -> {
133                 LOG.debug("{} => @Path {}", baseUri + clazz.getAnnotation(Path.class).value(), clazz);
134                 applicationContext.register(clazz);
135             });
136             reflections.getTypesAnnotatedWith(Provider.class).forEach(applicationContext::register);
137             reflections.getTypesAnnotatedWith(RequestScoped.class).forEach(applicationContext::register);
138             reflections.getTypesAnnotatedWith(ApplicationScoped.class).forEach(applicationContext::register);
139         } else {
140             LOG.debug("getClasses() = {}, not performing any scan", resourceClasses);
141             resourceClasses.forEach(applicationContext::register);
142         }
143 
144         deployment = new ResteasyDeployment();
145         deployment.setApplication(application);
146         deployment.start();
147 
148         final SpringBeanProcessor springBeanProcessor = new SpringBeanProcessor(deployment);
149 
150         applicationContext.addBeanFactoryPostProcessor(springBeanProcessor);
151         applicationContext.addApplicationListener(springBeanProcessor);
152         applicationContext.refresh();
153         if (baseApplicationContext != null) {
154             baseApplicationContext.getBeansWithAnnotation(Provider.class).forEach(
155                 (name,
156                     obj) -> {
157                     if (!deployment.getProviderFactory().isRegistered(obj)) {
158                         LOG.debug("registering {} into provider factory", name);
159                         deployment.getProviderFactory().register(obj);
160                     }
161                 });
162         }
163         applicationContext.getBeansWithAnnotation(Path.class).forEach((name,
164             obj) -> pathAnnotatedClasses.add(ClassUtils.getUserClass(obj)));
165 
166         httpClientOptions = applicationContext.getBean(HttpClientOptions.class);
167         if (httpClientOptions == null) {
168             httpClientOptions = new HttpClientOptions();
169         }
170         deployment.start();
171         dispatcher = deployment.getDispatcher();
172         providerFactory = deployment.getProviderFactory();
173     }

189     /**
190      * {@inheritDoc}
191      */
192     @Override
193     public void handle(final RoutingContext context) {
194 
195         final Client client = jaxRsClient(context.vertx());
196         final HttpServerRequest serverRequest = context.request();
197         final ResteasyUriInfo uriInfo = new ResteasyUriInfo(serverRequest.absoluteURI(), serverRequest.query(), baseUri.toASCIIString());
198 
199         final VertxHttpRequest request = new VertxHttpRequest(context, uriInfo, providerFactory);
200 
201         context.request().setExpectMultipart(isMultipartExpected(request));
202         try (final VertxHttpResponse response = new VertxHttpResponse(context)) {
203             context.vertx().executeBlocking(
204                 future -> {
205                     if (LOG.isDebugEnabled()) {
206                         LOG.debug("{} {}", context.request().method(), serverRequest.absoluteURI());
207                     }
208                     ThreadLocalResteasyProviderFactory.push(providerFactory);
209                     ResteasyProviderFactory.pushContext(RoutingContext.class, context);
210                     ResteasyProviderFactory.pushContext(Vertx.class, context.vertx());
211                     ResteasyProviderFactory.pushContext(Client.class, client);
212                     ResteasyProviderFactory.pushContext(HttpHeaders.class, request.getHttpHeaders());
213 
214                     try {
215                         dispatcher.invoke(request, response);
216                         context.response().end();
217                         future.complete();
218                     } finally {
219                         ResteasyProviderFactory.clearContextData();
220                         ThreadLocalResteasyProviderFactory.pop();
221                     }
222                 }, false,
223                 res -> {
224                     if (res.failed()) {
225                         context.fail(res.cause());
226                     }
227                 });
228         } catch (final IOException e) {
229             throw new UncheckedIOException(e);
230         }
231     }

233     /**
234      * Checks if multipart is expected for the resource. This is determined by the
235      * presence of {@value MediaType#MULTIPART_FORM_DATA} in the {@link Consumes}
236      * annotation.
237      *
238      * @param request
239      * @return
240      */
241     private boolean isMultipartExpected(final HttpRequest request) {
242 
243         final ResourceInvoker invoker = dispatcher.getRegistry().getResourceInvoker(request);
244         final Consumes consumes = invoker.getMethod().getAnnotation(Consumes.class);
245         return consumes != null && Arrays.asList(consumes.value()).contains(MediaType.MULTIPART_FORM_DATA);
246 
247     }

249     /**
250      * Get the JAX-RS Client that is present in the vertx context or create a new
251      * one.
252      *
253      * @param vertx
254      *            vertx
255      * @return JAX-RS client.
256      */
257     private Client jaxRsClient(final Vertx vertx) {
258 
259         Client jaxRsClient = vertx.getOrCreateContext().get(Client.class.getName());
260         if (jaxRsClient == null) {
261             final HttpClient httpClient = vertx.createHttpClient(httpClientOptions);
262             jaxRsClient = new ResteasyClientBuilder().providerFactory(providerFactory).httpEngine(new VertxClientEngine(httpClient)).build();
263             vertx.getOrCreateContext().put(Client.class.getName(), jaxRsClient);
264         }
265         return jaxRsClient;
266     }
