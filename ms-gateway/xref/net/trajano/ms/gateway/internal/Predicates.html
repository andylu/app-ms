3   import static io.vertx.core.http.HttpHeaders.AUTHORIZATION;
4   import static io.vertx.core.http.HttpHeaders.CONTENT_LENGTH;
5   import static io.vertx.core.http.HttpHeaders.COOKIE;
6   import static io.vertx.core.http.HttpHeaders.DATE;
7   import static io.vertx.core.http.HttpHeaders.SET_COOKIE;
8   import static io.vertx.core.http.HttpHeaders.TRANSFER_ENCODING;
9   import static net.trajano.ms.gateway.providers.RequestIDProvider.REQUEST_ID;
10  
11  import java.util.Map;
12  import java.util.Set;
13  import java.util.function.Predicate;
14  import java.util.stream.Collectors;
15  import java.util.stream.Stream;
16  
17  public final class Predicates {
18  
19      /**
20       * A header is can be forwarded as long as it is not restricted.
21       * {@link io.vertx.core.http.HttpHeaders#AUTHORIZATION} is restricted unless it
22       * is "Basic" which implies that it is a client authorization that can be passed
23       * down.
24       */
25      public static final Predicate<Map.Entry<String, String>> HEADER_FORWARDABLE;
26  
27      /**
28       * The {@link io.vertx.core.http.HttpHeaders#CONTENT_LENGTH} and
29       * {@link io.vertx.core.http.HttpHeaders#TRANSFER_ENCODING} causes problems when
30       * both are sent.
31       */
32      public static final Predicate<Map.Entry<String, String>> STRIP_CONTENT_LENGTH_AND_TRANSFER_ENCODING;
33  
34      /**
35       * Set of restricted headers.
36       */
37      private static final Set<String> RESTRICTED_HEADERS;
38  
39      /**
40       * URI that points to the JWKS.
41       */
42      private static final String X_JWKS_URI = "X-JWKS-URI";
43  
44      private static final String X_JWT_ASSERTION = "X-JWT-Assertion";
45  
46      private static final String X_JWT_AUDIENCE = "X-JWT-Audience";
47  
48      static {
49  
50          RESTRICTED_HEADERS = Stream.of(X_JWKS_URI, X_JWT_ASSERTION, X_JWT_AUDIENCE, REQUEST_ID, DATE, AUTHORIZATION, COOKIE, SET_COOKIE).map(CharSequence::toString).collect(Collectors.toSet());
51          HEADER_FORWARDABLE = e -> e.getKey().contentEquals(AUTHORIZATION) && e.getValue().startsWith("Basic ") || !RESTRICTED_HEADERS.contains(e.getKey());
52          STRIP_CONTENT_LENGTH_AND_TRANSFER_ENCODING = e -> !(e.getKey().contentEquals(CONTENT_LENGTH) || e.getKey().contentEquals(TRANSFER_ENCODING));
53  
54      }
55  
56      /**
57       * Prevent instantiation of utility class.
58       */
59      private Predicates() {
60  
61      }
62  }
